Express.js REST API

// Express.js REST API const express = require('express'); const app = express(); app.use(express.json()); app.get('/api/users', (req, res) => { res.json([ { id: 1, name: 'John Doe', email: 'john@example.com' }, { id: 2, name: 'Jane Smith', email: 'jane@example.com' } ]); }); app.post('/api/users', (req, res) => { const newUser = { id: Date.now(), ...req.body, createdAt: new Date() }; res.status(201).json(newUser); }); app.get('/api/users/:id', (req, res) => { const user = { id: parseInt(req.params.id), name: 'John Doe', email: 'john@example.com' }; res.json(user); }); const PORT = process.env.PORT || 3000; app.listen(PORT, () => { console.log(`Server running on port ${PORT}`); });
