Dataflow Overview

Basic

Dataflow is Google's unified stream and batch data processing service, based on Apache Beam. It provides a serverless, fully managed environment for running data processing pipelines that can handle both real-time streaming data and historical batch data.

⚡ Key Capabilities
  • Unified Model: Same code for batch and streaming
  • Auto-scaling: Dynamic worker allocation based on load
  • Fault-tolerant: Automatic error recovery and retries
  • Portable: Run on Google Cloud or other platforms

Processing Models

Basic

Dataflow supports both batch and streaming processing models with the same unified programming model. Choose the right model based on your data characteristics and latency requirements.

📦
Batch Processing
Process finite datasets with high throughput, suitable for historical data analysis
🌊
Streaming Processing

4def run_pipeline():
5 """Create and run a basic Dataflow pipeline."""
6 # Pipeline options
7 options = PipelineOptions(
8 project='my-project',
9 region='us-central1',
10 job_name='basic-pipeline',
11 temp_location='gs://my-bucket/temp',
12 staging_location='gs://my-bucket/staging'
13 )
14
15 with beam.Pipeline(options=options) as pipeline:
16 # Read from Cloud Storage
17 lines = pipeline | 'ReadFromGCS' >> beam.io.ReadFromText('gs://my-bucket/input/*.txt')
18
19 # Transform data
20 transformed = lines | 'SplitAndFilter' >> beam.Map(lambda line: line.split(',')[0]) \
21 | 'FilterEmpty' >> beam.Filter(lambda word: word.strip() != '')
22
23 # Count words
24 word_counts = transformed | 'CountWords' >> beam.combiners.Count.PerElement()
25
26 # Format results
27 formatted = word_counts | 'FormatResults' >> beam.Map(lambda wc: f'{wc[0]}: {wc[1]}')
28
29 # Write to BigQuery
30 formatted | 'WriteToBQ' >> beam.io.WriteToBigQuery(
31 table='my-project:my_dataset.word_counts',
32 schema='word:STRING, count:INTEGER',
33 write_disposition=beam.io.BigQueryDisposition.WRITE_TRUNCATE,
34 create_disposition=beam.io.BigQueryDisposition.CREATE_IF_NEEDED
35 )
36
37if __name__ == '__main__':

2from apache_beam import window
3from apache_beam.transforms import trigger

5class ParseEventFn(beam.DoFn):
6 def process(self, element):
7 """Parse event data from JSON string."""
8 import json
9 try:
10 event = json.loads(element)
11 yield {
12 'user_id': event['user_id'],
13 'event_type': event['event_type'],
14 'timestamp': event['timestamp'],
15 'value': event['value']
16 }
17 except json.JSONDecodeError:
18 pass # Skip invalid JSON
19
20def create_streaming_pipeline():
21 """Create a streaming pipeline with windowing."""
22 options = PipelineOptions(
23 streaming=True,
24 project='my-project',
25 region='us-central1'
26 )
27
28 with beam.Pipeline(options=options) as pipeline:
29 # Read from Pub/Sub
30 events = pipeline | 'ReadFromPubSub' >> beam.io.ReadFromPubSub(
31 subscription='projects/my-project/subscriptions/my-sub'
32 )
33
34 # Parse events
35 parsed_events = events | 'ParseEvents' >> beam.ParDo(ParseEventFn())
36
37 # Apply windowing (5-minute tumbling windows)
38 windowed_events = parsed_events | 'ApplyWindowing' >> beam.WindowInto(
39 window.FixedWindows(300) # 5 minutes
40 )
41
42 # Aggregate by user and event type
43 user_stats = windowed_events | 'GroupByUser' >> beam.GroupBy('user_id', 'event_type') \
44 | 'CountEvents' >> beam.combiners.Count.PerElement()
45
46 # Write to BigQuery
47 user_stats | 'WriteToBQ' >> beam.io.WriteToBigQuery(
48 table='my-project:analytics.user_events',
49 schema='user_id:STRING, event_type:STRING, count:INTEGER, window_start:TIMESTAMP'
50 )
51
52if __name__ == '__main__':
53 create_streaming_pipeline()

Windowing & Triggers

Advanced

Windowing allows you to group unbounded streaming data into finite chunks for processing. Combined with triggers, you can control when results are emitted for time-sensitive analytics.

1from apache_beam.transforms import trigger
2from apache_beam import window
3
4# Custom trigger for early results
5early_trigger = trigger.AfterWatermark(
6 early=trigger.AfterProcessingTime(30) # Early firing every 30s
7).withLateFirings(
8 trigger.AfterProcessingTime(300) # Late firings up to 5min
9)
10
11def create_windowed_pipeline():
12 """Pipeline with advanced windowing strategies."""
13
14 with beam.Pipeline() as pipeline:
15 events = pipeline | beam.io.ReadFromPubSub('projects/my-project/topics/events')
16
17 # Sliding windows with custom trigger
18 sliding_windowed = events | 'SlidingWindows' >> beam.WindowInto(
19 window.SlidingWindows(600, 60), # 10min windows, 1min slide
20 trigger=early_trigger,
21 accumulation_mode=trigger.AccumulationMode.ACCUMULATING
22 )
23
24 # Session windows for user sessions
25 session_windowed = events | 'SessionWindows' >> beam.WindowInto(
26 window.Sessions(1800), # 30min gap max
27 timestamp_combiner=window.TimestampCombiner.OUTPUT_AT_EOW
28 )
29
30 # Global window with custom trigger
31 global_windowed = events | 'GlobalWindow' >> beam.WindowInto(
32 window.GlobalWindows(),
33 trigger=trigger.AfterCount(1000) # Fire after 1000 elements
34 )
35
36 # Process each window type
37 results = [
38 sliding_windowed | 'ProcessSliding' >> beam.Map(process_sliding_window),
39 session_windowed | 'ProcessSessions' >> beam.Map(process_session_window),
40 global_windowed | 'ProcessGlobal' >> beam.Map(process_global_window)
41 ]
42
43 # Merge results
44 merged = results | beam.Flatten()
45
46 # Write results
47 merged | beam.io.WriteToBigQuery('my-project:analytics.windowed_results')
48
49def process_sliding_window(window_data):
50 """Process sliding window data."""
51 return {
52 'window_type': 'sliding',
53 'count': len(window_data),
54 'avg_value': sum(window_data) / len(window_data) if window_data else 0
55 }
56
57def process_session_window(window_data):
58 """Process session window data."""
59 return {
60 'window_type': 'session',
61 'session_length': max(window_data) - min(window_data),
62 'events': len(window_data)
63 }
64
65def process_global_window(window_data):
66 """Process global window data."""
67 return {
68 'window_type': 'global',
69 'batch_size': len(window_data),
70 'total_value': sum(window_data)
71 }
🪟 Windowing Types
  • Fixed Windows: Non-overlapping, fixed-size time windows
  • Sliding Windows: Overlapping windows for rolling aggregations
  • Session Windows: Dynamic windows based on activity gaps
  • Global Windows: Single window for entire pipeline lifetime
